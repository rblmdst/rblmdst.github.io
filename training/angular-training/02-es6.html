ES6 / ES2015

Purpose

Learn new Javascript features introduced by ES6.

LET

LET

let is similar to var but it is only accessible in the block level it is defined.


              if (true) {
                let a = 40;
                console.log(a); //40
              }
              console.log(a); // undefined

CONST

CONST

const is used to assign a constant value to the variable.


                const a = 50;
                a = 60; // shows error. You cannot change the value of const.
  
                const b = "Constant variable";
                b = "Assigning new value"; // shows error.

CONST

You cannot change the variable to reference some other memory location later.


            const LANGUAGES = ['Js', 'Ruby', 'Python', 'Go'];

            LANGUAGES = "Javascript"; // shows error. 
            
            LANGUAGES.push('Java'); // Works fine.
            console.log(LANGUAGES); // ['Js', 'Ruby', 'Python', 'Go', 'Java']

Template Strings

Template Strings


              // before
              var name = 'Sam';
              var age = 42;
              console.log('hello my name is ' + name + ' I am ' + age + ' years	old');

              // with template strings
              var name = 'Sam';
              var age = 42;
              console.log(`hello my name is ${name}, and I am ${age} years old`);

Arrow functions

Arrow functions


              // before es6
              function sum(a, b) {
                return a + b;
              }
              // with arrow functions
              var sum = (a, b) => {
                return a + b;
              }

              or

              var sum = (a, b) =>  a + b;

DEFAULT PARAMETER


            let func = (a, b = 10) => {
              return a + b;
            }
            func(20); // 20 + 10 = 30

            func(20, 50); // 20 + 50 = 70

REST AND SPREAD OPERATORS

REST AND SPREAD OPERATORS


                // function accept three params
                const sum = (a, b, c) {
                  return a + b + c;
                }
                sum(1, 4, 5); // 10
  
                const vals = [1, 4, 5];
                // sum(...vals)  <=>  sum(vals[0], vals[1], vals[2])
                sum(...vals); // 10

                // sum(10, ...vals)  <=>  sum(10, vals[0], vals[1])
                sum(10, ...vals); // 15

REST AND SPREAD OPERATORS


                Math.max(10, 20, 60, 100, 50, 200); // returns 200

                let arr = [10, 20, 60];
                Math.max(arr); // Shows error. Doesn't accept an array
                Math.max(...arr); // 60

DESTRUCTURING

Destructuring

Destructuring is a way to quickly extract data out of an object or array with less code.

Array Destructuring


              // before
              const foo	= ['one', 'two', 'three'];
              const one = foo[0];
              const two = foo[1];
              const three = foo[2];

              // with destructuring
              const foo = ['one', 'two', 'three'];
              const [one, two, three] = foo;
              console.log(one); // 'one'

Object Destructuring


                // before
                let user = {username: 'John', age: 10, sex: 'M'};
                let username = user.username;
                let age = user.age;

                // with Destructuring
                const {username, age} = user;
                console.log(username); // 'John'
                console.log(age); // 10

CLASSES

Declaration


            class Person {
              /* attributes/properties can be public or private */
              private _name;
              constructor(name) {
                this._name = name;
              }

              /* methods can also can be public or private */
              public introducedMe() {
                console.log(`Hello, I am ${this._name}.`);
              }
            }

Instanciation


              let person1 = new Person('John Doe'); // create a person
              person1.introducedMe(); // Hello, I am John Doe.

MODULES

A module is just a simple Javascript file with import/export statements.


              // math-utils.js
              export const PI = 3.14;
              export const square = (x) => x * x;

              // index.js
              import { square } from './math-utils';
              console.log(square(4)); // 16

  • Use export to export variables or functions that can be accessible outside of the module.
  • Use import to import variables or functions from other modules.
